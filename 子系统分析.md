# spi

驱动框架大体分为三层（除开硬件和用户态）：

1. 驱动层
2. 核心层
3. 控制器驱动层

## 核心层

代码位于`drivers/spi`

`spi.c`: 

1. 对子系统进行初始化工作,主要注册`spi_bus`,`spi_master_class`并提供`spi`设备驱动对总线操作的API
2. 对控制器层提供注册各厂家控制器的api和回调函数

`spi.h`包含了一些重要的数据结构,`struct spi_master; struct spi_transfer; struct spi_message`以及一些简单的函数

`spi-gpio.c`: 一个框架允许使用gpio模拟spi

`spi-bitbang.c`：是Linux内核中提供的一个通用框架，用于在没有硬件SPI控制器或需要灵活控制SPI时序和配置的系统中模拟SPI总线的通信

核心层起到承上启下的作用,对设备驱动层提供spi的标准api,对控制器驱动层, 提供注册控制器接口,并提供一些回调函数

### 核心层初始化

`spi_init`函数: 主要负责注册spi总线, 以及spi主机类, 可以分别的在`/sys/bus/spi`和`sys/class/spi_master`找到

```c
static int __init spi_init(void) {
	int	status;
	buf = kmalloc(SPI_BUFSIZ, GFP_KERNEL);
	if (!buf) {
		status = -ENOMEM;
		goto err0;
	}
	status = bus_register(&spi_bus_type);
	if (status < 0)
		goto err1;
	status = class_register(&spi_master_class);
	if (status < 0)
		goto err2;
	if (IS_ENABLED(CONFIG_SPI_SLAVE)) {
		status = class_register(&spi_slave_class);
		if (status < 0)
			goto err3;
	}

	if (IS_ENABLED(CONFIG_OF_DYNAMIC))
		WARN_ON(of_reconfig_notifier_register(&spi_of_notifier));
	if (IS_ENABLED(CONFIG_ACPI))
		WARN_ON(acpi_reconfig_notifier_register(&spi_acpi_notifier));

	return 0;
err3:
	class_unregister(&spi_master_class);
err2:
	bus_unregister(&spi_bus_type);
err1:
	kfree(buf);
	buf = NULL;
err0:
	return status;
}
```



**当设备与驱动匹配**之后,就会调用`spi_match_device`函数, 也就是spi总线上的match函数

```c
struct bus_type spi_bus_type = {
	.name		= "spi",
	.dev_groups	= spi_dev_groups,
	.match		= spi_match_device,
	.uevent		= spi_uevent,
};

static int spi_match_device(struct device *dev, struct device_driver *drv)
{
	const struct spi_device	*spi = to_spi_device(dev);
	const struct spi_driver	*sdrv = to_spi_driver(drv);

	/* Attempt an OF style match */
	if (of_driver_match_device(dev, drv))
		return 1;			//通过设备树匹配

	/* Then try ACPI */
	if (acpi_driver_match_device(dev, drv))
		return 1;				//通过acpi匹配

	if (sdrv->id_table)
		return !!spi_match_id(sdrv->id_table, spi);		//通过id_table匹配

	return strcmp(spi->modalias, drv->name) == 0;		//通过name匹配
}
```

#### 控制器层



 `spi_register_master`给控制器层提供

```c
int spi_register_master(struct spi_master *master)			//这里的master是通过spi_alloc_master()获得的
{
	static atomic_t		dyn_bus_id = ATOMIC_INIT((1<<15) - 1);
	struct device		*dev = master->dev.parent;
	struct boardinfo	*bi;
	int			status = -ENODEV;
	int			dynamic = 0;

	if (!dev)
		return -ENODEV;
	/*
	根据设备树中的cs-gpios,向master添加cs引脚,把当前master的所有cs引脚都添加进
	*/
	status = of_spi_register_master(master);	
	if (status)
		return status;

	/* even if it's just one always-selected device, there must
	 * be at least one chipselect
	 */
	if (master->num_chipselect == 0)
		return -EINVAL;

	if ((master->bus_num < 0) && master->dev.of_node)
		master->bus_num = of_alias_get_id(master->dev.of_node, "spi");

	/* convention:  dynamically assigned bus IDs count down from the max */
	if (master->bus_num < 0) {
		/* FIXME switch to an IDR based scheme, something like
		 * I2C now uses, so we can't run out of "dynamic" IDs
		 */
		master->bus_num = atomic_dec_return(&dyn_bus_id);
		dynamic = 1;
	}

	INIT_LIST_HEAD(&master->queue);
	spin_lock_init(&master->queue_lock);
	spin_lock_init(&master->bus_lock_spinlock);
	mutex_init(&master->bus_lock_mutex);
	master->bus_lock_flag = 0;
	init_completion(&master->xfer_completion);
	if (!master->max_dma_len)
		master->max_dma_len = INT_MAX;

	/* register the device, then userspace will see it.
	 * registration fails if the bus ID is in use.
	 */
	dev_set_name(&master->dev, "spi%u", master->bus_num);
	status = device_add(&master->dev);		//将device注册到设备(master)模型中
	if (status < 0)
		goto done;
	dev_dbg(dev, "registered master %s%s\n", dev_name(&master->dev),
			dynamic ? " (dynamic)" : "");

	/* If we're using a queued driver, start the queue */
	if (master->transfer)
		dev_info(dev, "master is unqueued, this is deprecated\n");
	else {
        //master没实现自己的transfer函数,则初始化发送队列, 核心层填充默认的transfer函数 
		status = spi_master_initialize_queue(master);		
        
		if (status) {
			device_del(&master->dev);
			goto done;
		}
	}

	mutex_lock(&board_lock);
	list_add_tail(&master->list, &spi_master_list);
	list_for_each_entry(bi, &board_list, list)
		spi_match_master_to_boardinfo(master, &bi->board_info);//注册spi_device老方法
	mutex_unlock(&board_lock);

	/* Register devices from the device tree and ACPI */
	of_register_spi_devices(master);					//以下是两个注册spi_device新方法
	acpi_register_spi_devices(master);					//把设备树里描述的 SPI 外设，‘挂到’对应的 SPI 控制器总线上
done:
	return status;
}

//初始化发送队列函数,填充master的成员函数
static int spi_master_initialize_queue(struct spi_master *master)
{
	int ret;

	master->transfer = spi_queued_transfer;
	if (!master->transfer_one_message)
		master->transfer_one_message = spi_transfer_one_message;

	/* Initialize and start queue */
	ret = spi_init_queue(master);			//初始化内核工作线程,设置执行函数
	if (ret) {
		dev_err(&master->dev, "problem initializing queue\n");
		goto err_init_queue;
	}
	master->queued = true;
	ret = spi_start_queue(master);			//唤醒工作线程  到这里队列化的相关工作完成,等待message请求,在线程中处理传送任务
	if (ret) {
		dev_err(&master->dev, "problem starting queue\n");
		goto err_start_queue;
	}

	return 0;

err_start_queue:
	spi_destroy_queue(master);
err_init_queue:
	return ret;
}

```

```c
int spi_async(struct spi_device *spi, struct spi_message *message)
{
	struct spi_master *master = spi->master;
	int ret;
	unsigned long flags;

	ret = __spi_validate(spi, message);
	if (ret != 0)
		return ret;

	spin_lock_irqsave(&master->bus_lock_spinlock, flags);

	if (master->bus_lock_flag)
		ret = -EBUSY;
	else
		ret = __spi_async(spi, message);

	spin_unlock_irqrestore(&master->bus_lock_spinlock, flags);

	return ret;
}
```

`spi_sync`

```
static int __spi_sync(struct spi_device *spi, struct spi_message *message,
		      int bus_locked)
{
	DECLARE_COMPLETION_ONSTACK(done);
	int status;
	struct spi_master *master = spi->master;
	unsigned long flags;

	status = __spi_validate(spi, message);
	if (status != 0)
		return status;

	message->complete = spi_complete;
	message->context = &done;
	message->spi = spi;

	if (!bus_locked)
		mutex_lock(&master->bus_lock_mutex);

	/* If we're not using the legacy transfer method then we will
	 * try to transfer in the calling context so special case.
	 * This code would be less tricky if we could remove the
	 * support for driver implemented message queues.
	 */
	if (master->transfer == spi_queued_transfer) {
		spin_lock_irqsave(&master->bus_lock_spinlock, flags);

		trace_spi_message_submit(message);

		status = __spi_queued_transfer(spi, message, false);

		spin_unlock_irqrestore(&master->bus_lock_spinlock, flags);
	} else {
		status = spi_async_locked(spi, message);
	}

	if (!bus_locked)
		mutex_unlock(&master->bus_lock_mutex);

	if (status == 0) {
		/* Push out the messages in the calling context if we
		 * can.
		 */
		if (master->transfer == spi_queued_transfer)
			__spi_pump_messages(master, false);

		wait_for_completion(&done);
		status = message->status;
	}
	message->context = NULL;
	return status;
}
```

# i2c

从上至下分为三层结构

设备驱动层: 使用i2c子系统

i2c核心层: 提供核心api,把client(类似spi_device)设备挂载到总线上, 维护i2c_driver和i2c_client链表, 实现二者的匹配

适配层: 底层控制器驱动, 实现i2c时序,  实现i2c总线发送和接收数据的方法

```c
	 * using this, it will be removed in a near future.

	int (*probe)(struct i2c_client *, const struct i2c_device_id *);
	int (*remove)(struct i2c_client *);

	/* New driver model interface to aid the seamless removal of the
	 * current probe()'s, more commonly unused than used second parameter.

	void (*shutdown)(struct i2c_client *);

	 * The format and meaning of the data value depends on the protocol.
	 * For the SMBus alert protocol, there is a single bit of data passed
	 * as the alert response's low bit ("event flag").
	 * For the SMBus Host Notify protocol, the data corresponds to the

		      unsigned int data); // 警告回调函数（例如SMBus警报协议）
```

   ```c
   #define module_driver(__driver, __register, __unregister, ...) \
   static int __init __driver##_init(void) \
   { \
   	return __register(&(__driver) , ##__VA_ARGS__); \
   } \
   module_init(__driver##_init); \
   static void __exit __driver##_exit(void) \
   { \
   	__unregister(&(__driver) , ##__VA_ARGS__); \
   } \
   module_exit(__driver##_exit);
   
   /**
    * builtin_driver() - Helper macro for drivers that don't do anything
    * special in init and have no exit. This eliminates some boilerplate.
    * Each driver may only use this macro once, and calling it replaces
    * device_initcall (or in some cases, the legacy __initcall).  This is
    * meant to be a direct parallel of module_driver() above but without
    * the __exit stuff that is not used for builtin cases.
    *
    * @__driver: driver name
    * @__register: register function for this driver type
    * @...: Additional arguments to be passed to __register
    *
    * Use this macro to construct bus specific macros for registering
    * drivers, and do not use it on its own.
    */
   #define builtin_driver(__driver, __register, ...) \
   static int __init __driver##_init(void) \
   { \
   	return __register(&(__driver) , ##__VA_ARGS__); \
   } \
   device_initcall(__driver##_init);
   ```

## 驱动流程

### 核心层注册

可以看到内核启动阶段就调用init函数,注册了i2c总线,和设备驱动dummy_driver

```c
//i2c-core.c
struct bus_type i2c_bus_type = {
	.name		= "i2c",
	.match		= i2c_device_match,	//总线上驱动和设备的匹配函数
	.probe		= i2c_device_probe,	//总线上设备和驱动匹配时调用
	.remove		= i2c_device_remove,
	.shutdown	= i2c_device_shutdown,
	.pm		= &i2c_device_pm_ops,
};
EXPORT_SYMBOL_GPL(i2c_bus_type);
static const struct i2c_device_id dummy_id[] = {
	{ "dummy", 0 },
	{ },
};
static struct i2c_driver dummy_driver = {
	.driver.name	= "dummy",
	.probe		= dummy_probe,
	.remove		= dummy_remove,
	.id_table	= dummy_id,
};
static int __init i2c_init(void) {
	int retval;
	//注册I2C总线
	retval = bus_register(&i2c_bus_type);
	if (retval)
		return retval;	
	//向I2C总线注册一个名字为dummy的驱动，这个驱动没什么实际的功能，空实现的驱动
	retval = i2c_add_driver(&dummy_driver);
	if (retval)
		goto class_err;
	return 0;
class_err:
	bus_unregister(&i2c_bus_type);
	return retval;
}
static void __exit i2c_exit(void) {
	//删除dummy驱动
	i2c_del_driver(&dummy_driver);
	//卸载I2C总线
	bus_unregister(&i2c_bus_type);
}

/* We must initialize early, because some subsystems register i2c drivers
 * in subsys_initcall() code, but are linked (and initialized) before i2c.
 */
postcore_initcall(i2c_init);
module_exit(i2c_exit);

```

- 匹配函数

  ```c
  static int i2c_device_match(struct device *dev, struct device_driver *drv) {
  	struct i2c_client *client = i2c_verify_client(dev);
  	struct i2c_driver *driver;
  	if (!client)
  		return 0;
  	/* Attempt an OF style match */
  	if (of_driver_match_device(dev, drv))
  		return 1;
  	/* Then ACPI style match */
  	if (acpi_driver_match_device(dev, drv))
  		return 1;
  	driver = to_i2c_driver(drv);
  	/* match on an id table if there is one */
  	if (driver->id_table)
  		return i2c_match_id(driver->id_table, client) != NULL;
  	return 0;
  }
  
  ```

  匹配之后就会调用probe函数, I2C总线的probe方法就是进一步调用i2c_driver的probe方法。

  ```c
  static int i2c_device_probe(struct device *dev) {
  	//利用container_of宏获取到I2C设备结构体
  	struct i2c_client	*client = i2c_verify_client(dev);	//检查client的合法性
  	struct i2c_driver	*driver;
  	int status;
  
  	if (!client)
  		return 0;
  
  	//利用container_of宏获取到I2C驱动结构体
  	driver = to_i2c_driver(dev->driver);
  	if (!driver->probe || !driver->id_table)
  		return -ENODEV;
  		
  	// 把I2C设备驱动和I2C设备绑定，将来可以互相查找到对方
  	client->driver = driver;
  	if (!device_can_wakeup(&client->dev))
  		device_init_wakeup(&client->dev,
  					client->flags & I2C_CLIENT_WAKE);
  	dev_dbg(dev, "probe\n");
  
  	//调用I2C驱动的probe函数
  	status = driver->probe(client, i2c_match_id(driver->id_table, client));//设备树匹配时,probe的第二个参数为NULL
  	if (status) {
  		client->driver = NULL;
  		i2c_set_clientdata(client, NULL);
  	}
  	return status;
  }
  ```

```c
static int i2c_imx_xfer(struct i2c_adapter *adapter,
			struct i2c_msg *msgs, int num) {
	struct imx_i2c_struct *i2c_imx = i2c_get_adapdata(adapter);
	int result;
	result = pm_runtime_get_sync(i2c_imx->adapter.dev.parent);
	if (result < 0)
		return result;

	result = i2c_imx_xfer_common(adapter, msgs, num, false);
	pm_runtime_mark_last_busy(i2c_imx->adapter.dev.parent);
	pm_runtime_put_autosuspend(i2c_imx->adapter.dev.parent);
	return result;
}

			       struct i2c_msg *msgs, int num) {

	int result;
	result = clk_enable(i2c_imx->clk);
	if (result)
		return result;

	result = i2c_imx_xfer_common(adapter, msgs, num, true);
	clk_disable(i2c_imx->clk);
	return result;
}
```

`i2c_imx_start`、`i2c_imx_read`、`i2c_imx_dma_write/i2c_imx_write` 和 `i2c_imx_stop` 这些函数就是 I2C 寄存器的具体操作函数，按照i2c协议

```c
static int i2c_imx_xfer_common(struct i2c_adapter *adapter, struct i2c_msg *msgs, int num, bool atomic) {
	unsigned int i, temp;
	int result;
	bool is_lastmsg = false;
	struct imx_i2c_struct *i2c_imx = i2c_get_adapdata(adapter);

	dev_dbg(&i2c_imx->adapter.dev, "<%s>\n", __func__);

	/* Start I2C transfer */
	result = i2c_imx_start(i2c_imx, atomic);
	if (result) {
		/*
		 * Bus recovery uses gpiod_get_value_cansleep() which is not
		 * allowed within atomic context.
		 */
		if (!atomic && i2c_imx->adapter.bus_recovery_info) {
			i2c_recover_bus(&i2c_imx->adapter);
			result = i2c_imx_start(i2c_imx, atomic);
		}
	}

	if (result)
		goto fail0;
	/* read/write data */
	for (i = 0; i < num; i++) {
		if (i == num - 1)
			is_lastmsg = true;

		if (i) {
			dev_dbg(&i2c_imx->adapter.dev,
				"<%s> repeated start\n", __func__);
			temp = imx_i2c_read_reg(i2c_imx, IMX_I2C_I2CR);
			temp |= I2CR_RSTA;
			imx_i2c_write_reg(temp, i2c_imx, IMX_I2C_I2CR);
			result = i2c_imx_bus_busy(i2c_imx, 1, atomic);
			if (result)
				goto fail0;
		}
		dev_dbg(&i2c_imx->adapter.dev,
			"<%s> transfer message: %d\n", __func__, i);
		/* write/read data */
#ifdef CONFIG_I2C_DEBUG_BUS
		temp = imx_i2c_read_reg(i2c_imx, IMX_I2C_I2CR);
		dev_dbg(&i2c_imx->adapter.dev,
			"<%s> CONTROL: IEN=%d, IIEN=%d, MSTA=%d, MTX=%d, TXAK=%d, RSTA=%d\n",
			__func__,
			(temp & I2CR_IEN ? 1 : 0), (temp & I2CR_IIEN ? 1 : 0),
			(temp & I2CR_MSTA ? 1 : 0), (temp & I2CR_MTX ? 1 : 0),
			(temp & I2CR_TXAK ? 1 : 0), (temp & I2CR_RSTA ? 1 : 0));
		temp = imx_i2c_read_reg(i2c_imx, IMX_I2C_I2SR);
		dev_dbg(&i2c_imx->adapter.dev,
			"<%s> STATUS: ICF=%d, IAAS=%d, IBB=%d, IAL=%d, SRW=%d, IIF=%d, RXAK=%d\n",
			__func__,
			(temp & I2SR_ICF ? 1 : 0), (temp & I2SR_IAAS ? 1 : 0),
			(temp & I2SR_IBB ? 1 : 0), (temp & I2SR_IAL ? 1 : 0),
			(temp & I2SR_SRW ? 1 : 0), (temp & I2SR_IIF ? 1 : 0),
			(temp & I2SR_RXAK ? 1 : 0));
#endif
		if (msgs[i].flags & I2C_M_RD) {
			result = i2c_imx_read(i2c_imx, &msgs[i], is_lastmsg, atomic);
		} else {
			if (!atomic &&
			    i2c_imx->dma && msgs[i].len >= DMA_THRESHOLD)
				result = i2c_imx_dma_write(i2c_imx, &msgs[i]);
			else
				result = i2c_imx_write(i2c_imx, &msgs[i], atomic);
		}
		if (result)
			goto fail0;
	}
fail0:
	/* Stop I2C transfer */
	i2c_imx_stop(i2c_imx, atomic);

	dev_dbg(&i2c_imx->adapter.dev, "<%s> exit with: %s: %d\n", __func__,
		(result < 0) ? "error" : "success msg",
			(result < 0) ? result : num);
	return (result < 0) ? result : num;
}
```
